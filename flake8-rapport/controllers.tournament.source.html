7 def generate_pairs_first_round(rnd, trn_players):
8 """Create pairs for the first round (swiss rules)."""
9 players = sorted(trn_players, key=lambda x: x.rank)
10 half = len(players) // 2
11 upper_list = players[:half]
12 lower_list = players[half:]
13 for i, j in zip(range(len(upper_list)), range(len(lower_list))):
14 rnd.games.append(Game(upper_list[i], lower_list[j]))
